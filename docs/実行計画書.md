### 4-2. DOMContentLoaded 内の処理を書き換え

```javascript
document.addEventListener('DOMContentLoaded', () => {
    // ----- サイドバー -----
    document.getElementById('brand-logo').textContent = config.name;
    document.getElementById('sidebar-desc').textContent = config.tagline;
    document.getElementById('sidebar-hours').textContent = config.businessHours;
    document.getElementById('sidebar-gbp-link').href = config.gbpUrl;
    
    // ----- ヒーロー -----
    const heroSection = document.getElementById('hero');
    if (heroSection) {
        heroSection.style.backgroundImage = `url('${config.heroImage}')`;
    }
    document.getElementById('hero-title').textContent = config.heroTitle;
    document.getElementById('hero-copy').textContent = config.heroCopy;
    
    // ----- メニュー -----
    const menuGrid = document.getElementById('menu-grid');
    if (menuGrid) {
        config.menuItems.forEach(item => {
            const div = document.createElement('div');
            div.className = 'menu-item';
            div.innerHTML = `
                <img src="${item.image}" alt="${item.name}" class="menu-img" loading="lazy" width="300" height="300">
                <span class="menu-name">${item.name}</span>
                <span class="menu-price">${item.price}</span>
            `;
            menuGrid.appendChild(div);
        });
    }
    
    // ----- 店内雰囲気 -----
    const interiorImg = document.getElementById('interior-img');
    if (interiorImg) {
        interiorImg.src = config.interiorImage;
    }
    const interiorText = document.getElementById('interior-text');
    if (interiorText) {
        interiorText.textContent = config.interiorText;
    }
    
    // ----- オーナー -----
    const ownerImg = document.getElementById('owner-img');
    if (ownerImg) {
        ownerImg.src = config.ownerImage;
    }
    document.getElementById('owner-name').textContent = config.ownerName;
    document.getElementById('owner-text').textContent = config.ownerProfile;
    
    // ----- アクセス -----
    document.getElementById('access-address').textContent = config.address;
    document.getElementById('access-parking').textContent = config.parking;
    document.getElementById('access-hours').textContent = config.regularHours;
    document.getElementById('access-map-link').href = config.mapUrl;
    
    // ----- フッター -----
    document.getElementById('footer-instagram').href = config.instagramUrl;
    document.getElementById('footer-gbp').href = config.gbpUrl;
});
```

## 補足

- Instagram埋め込みエリアはプレースホルダーとして `#access-embed` を用意。後から埋め込みコードまたは別の内容に差し替え可能
- 画像がまだない場合は、プレースホルダー画像スキルで仮画像を生成可能
- コピーテキスト（キャッチコピー、オーナー紹介文など）は config 内で一元管理されているため、後から簡単に変更可能
